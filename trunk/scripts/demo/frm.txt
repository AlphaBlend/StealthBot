Sub frm_Initialize()

   frm.CreateObj "TextBox", "text"

   With frm.GetObjByName("text")
      .Visible = True
   End With

End Sub

Sub frm_Load()

   AddChat vbBlue, "frm_Load()"

End Sub

Sub frm_text_Change()

   AddChat vbBlue, "frm_text_Change()"

End Sub

Sub frm_Unload()

   frm.ClearObjs

End Sub
