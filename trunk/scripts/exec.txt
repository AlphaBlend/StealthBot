Option Explicit

' // script data
Sub Data()

   Script("Name") = "Demo"
   Script("Author") = "Eric[nK]"
   Script("Major") = 1
   Script("Minor") = 0
   Script("Revision") = 1

End Sub

Sub Event_PressedEnter(Text)

   ' check for "exec" command
   If (Left(LCase(Text), 5) = "/exec") Then
      Dim i    ' loop counter
      Dim code ' stores command code

      ' prevent internal processing
      VetoThisMessage  

      ' pull code from command
      code = Mid(Text, 7)

      ' call code locally
      ExecuteGlobal code
   End If
   
End Sub
